## Installation

<Steps>
  <Step>
    ### Install dependencies
    
    ```bash
    npm install motion clsx tailwind-merge
    ```
  </Step>
      <Step>
        ### Add util file
        
        Create `lib/utils.ts` if you haven't already.

        ```ts title="lib/utils.ts"
        import { type ClassValue, clsx } from 'clsx';
        import { twMerge } from 'tailwind-merge';

        export function cn(...inputs: ClassValue[]) {
          return twMerge(clsx(inputs));
        }
        ```
      </Step>
      <Step>
        ### Copy the source code
        
        Copy the code into `app/components/TextBlocks/BluxText.tsx`.

        ```tsx title="app/components/TextBlocks/BluxText.tsx"
        'use client';

        import { cn } from '@/lib/utils';
        import { motion, type HTMLMotionProps } from 'motion/react';
        import React, { useState } from 'react';

        interface BlurTextProps extends HTMLMotionProps<'div'> {
          children: React.ReactNode;
          blur?: string; // e.g. "10px"
          hoverToReveal?: boolean;
          clickToReveal?: boolean;
          revealDuration?: number;
          className?: string;
        }

        export function BlurText({
          children,
          blur = '8px',
          hoverToReveal = false,
          clickToReveal = false,
          revealDuration = 0.3,
          className,
          animate,
          ...props
        }: BlurTextProps) {
          const [isRevealed, setIsRevealed] = useState(false);
          const isInteractive = hoverToReveal || clickToReveal;

          const handleHoverStart = () => {
            if (hoverToReveal) setIsRevealed(true);
          };

          const handleHoverEnd = () => {
             if (hoverToReveal) setIsRevealed(false);
          };

          const handleClick = () => {
            if (clickToReveal) setIsRevealed(!isRevealed);
          };

          const variants = {
            blurred: {
              filter: `blur(${blur})`,
              opacity: 0.8,
            },
            revealed: {
              filter: 'blur(0px)',
              opacity: 1,
            },
          };

          return (
            <motion.div
              className={cn('inline-block cursor-default relative', className, {
                'cursor-pointer': clickToReveal,
              })}
              initial="blurred"
              animate={isInteractive ? (isRevealed ? 'revealed' : 'blurred') : 'blurred'}
              variants={variants}
              transition={{ duration: revealDuration, ease: 'easeOut' }}
              onHoverStart={handleHoverStart}
              onHoverEnd={handleHoverEnd}
              onClick={handleClick}
              {...props}
            >
              {children}
            </motion.div>
          );
        }
        ```
  </Step>
</Steps>

## Examples

### Static Blur (Default)

The text remains blurred unless customized.

<Preview>
  <BlurText>
    Hidden wisdom lies within this text.
  </BlurText>
</Preview>

### Reveal on Hover

Pass `hoverToReveal` to clear the blur when the mouse enters.

<Preview>
  <BlurText hoverToReveal>
    Hover over me to reveal the secret message!
  </BlurText>
</Preview>

### Reveal on Click

Pass `clickToReveal` to toggle the blur on click.

<Preview>
  <BlurText clickToReveal>
    Click me to toggle visibility.
  </BlurText>
</Preview>

### Custom Blur Intensity

Adjust the `blur` prop to change the intensity (default is `8px`).

<Preview className="flex-col space-y-4">
  <BlurText blur="4px" hoverToReveal>
    Light blur (4px)
  </BlurText>
  
  <BlurText blur="12px" hoverToReveal>
    Heavy blur (12px)
  </BlurText>
</Preview>

### Reveal Duration

Control the transition speed with `revealDuration` (default `0.3`s).

<Preview>
  <BlurText hoverToReveal revealDuration={1}>
    Slow reveal (1s duration)...
  </BlurText>
</Preview>

## Props

| Prop | Type | Default | Description |
| :--- | :--- | :--- | :--- |
| `children` | `ReactNode` | - | The content to be blurred. |
| `blur` | `string` | `'8px'` | CSS blur value (e.g., '10px'). |
| `hoverToReveal` | `boolean` | `false` | Whether to reveal text on hover. |
| `clickToReveal` | `boolean` | `false` | Whether to toggle text visibility on click. |
| `revealDuration` | `number` | `0.3` | Duration of the reveal animation in seconds. |
| `className` | `string` | - | Additional CSS classes. |

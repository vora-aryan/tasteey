## Installation

<Steps>
  <Step>
    ### Install dependencies
    
    ```bash
    npm install clsx tailwind-merge
    ```
  </Step>
  <Step>
    ### Add Fonts
    
    Add the necessary fonts to your `app/root.tsx` or `layout.tsx`:

    ```tsx
    export const links: LinksFunction = () => [
      { rel: "preconnect", href: "https://fonts.googleapis.com" },
      { rel: "preconnect", href: "https://fonts.gstatic.com", crossOrigin: "anonymous" },
      { href: "https://fonts.googleapis.com/css2?family=Amatic+SC:wght@400;700&family=Just+Another+Hand&display=swap", rel: "stylesheet" },
    ];
    ```
  </Step>
  <Step>
    ### Add CSS Utilities
    
    Add custom font utilities to your CSS (e.g., `app/app.css`):

    ```css
    .font-hand-amatic {
      font-family: 'Amatic SC', cursive;
    }
    .font-hand-just {
      font-family: 'Just Another Hand', cursive;
    }
    ```
  </Step>
  <Step>
    ### Copy the source code
    
    Copy the code into `app/components/TextBlocks/HandWrittenText.tsx` (for TS) or `app/components/TextBlocks/HandWrittenText.jsx` (for JS).

    <Tabs items={["TS", "JS"]}>
      <Tab value="TS">
        ```tsx title="app/components/TextBlocks/HandWrittenText.tsx"
        import { cn } from "@/lib/utils";

        interface HandWrittenTextProps {
          children: React.ReactNode;
          className?: string;
          type?: "amatic" | "just";
          as?: React.ElementType;
          color?: string;
          space?: number; // Letter spacing in rem
          aspectRatio?: "1/1" | "16/9" | "4/3" | "3/4" | "21/9";
          image?: string; // Custom background image URL
        }

        export function HandWrittenText({
          children,
          className,
          type = "just",
          as: Component = "p",
          color = "#0036cbff", // Default ballpoint pen blue
          space = 0.05,
          aspectRatio,
          image = "/paper-texture.jpg",
        }: HandWrittenTextProps) {
          const fontClass = {
            amatic: "font-hand-amatic",
            just: "font-hand-just",
          };

          const aspectStyles = {
            "1/1": { aspectRatio: "1 / 1" },
            "16/9": { aspectRatio: "16 / 9" },
            "4/3": { aspectRatio: "4 / 3" },
            "3/4": { aspectRatio: "3 / 4" },
            "21/9": { aspectRatio: "21 / 9" },
          };

          return (
            <Component
              className={cn(
                "text-2xl p-8 rounded-lg shadow-md",
                fontClass[type],
                className
              )}
              style={{
                backgroundImage: `url('${image}')`,
                backgroundSize: "cover",
                backgroundPosition: "center",
                color: color,
                letterSpacing: `${space}em`,
                textShadow: "0.5px 0.5px 0px rgba(0,0,0,0.3)", // Subtle ink depth
                ...(aspectRatio ? aspectStyles[aspectRatio] : {}),
              }}
            >
              {children}
            </Component>
          );
        }
        ```
      </Tab>

## Props

| Prop | Type | Default | Description |
| :--- | :--- | :--- | :--- |
| `children` | `ReactNode` | - | The text content. |
| `type` | `'amatic' \| 'just'` | `'just'` | The handwriting style to use. |
| `aspectRatio` | `'1/1' \| '16/9' \| '4/3' \| '3/4' \| '21/9'` | - | Optional aspect ratio for the paper container. |
| `image` | `string` | `'/paper-texture.jpg'` | Custom background image URL. |
| `color` | `string` | `'#0036cbff'` | Custom text color. Defaults to ballpoint blue. |
| `space` | `number` | `0.05` | Character spacing in `em`. |
| `className` | `string` | - | Additional CSS classes. |
| `as` | `React.ElementType` | `p` | The HTML element to render. |

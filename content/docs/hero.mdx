## Installation

<Steps>
  <Step>
    ### Install dependencies
    
    No external dependencies required (uses standard Tailwind CSS).
  </Step>
  <Step>
    ### Copy the source code
        
        Copy the code into `app/components/Hero.tsx`.

        ```tsx title="app/components/Hero.tsx"
        import { type ReactNode } from "react";

        interface HeroProps {
          title: string;
          subtitle?: string;
          primaryCta?: {
            label: string;
            href: string;
          };
          secondaryCta?: {
            label: string;
            href: string;
          };
          children?: ReactNode;
        }

        export function Hero({
          title,
          subtitle,
          primaryCta,
          secondaryCta,
          children,
        }: HeroProps) {
          return (
            <div className="relative overflow-hidden rounded-3xl bg-zinc-950 px-6 py-24 sm:py-32 isolate">
              {/* Background Gradients */}
              <div className="absolute inset-0 -z-10 bg-[radial-gradient(45rem_50rem_at_top,theme(colors.zinc.800),theme(colors.zinc.950))]" />
              <div className="absolute inset-x-0 -top-40 -z-10 transform-gpu overflow-hidden blur-3xl sm:-top-80">
                <div
                  className="relative left-[calc(50%-11rem)] aspect-[1155/678] w-[36.125rem] -translate-x-1/2 rotate-[30deg] bg-gradient-to-tr from-zinc-200 to-zinc-600 opacity-20 sm:left-[calc(50%-30rem)] sm:w-[72.1875rem]"
                  style={{
                    clipPath:
                      "polygon(74.1% 44.1%, 100% 61.6%, 97.5% 26.9%, 85.5% 0.1%, 80.7% 2%, 72.5% 32.5%, 60.2% 62.4%, 52.4% 68.1%, 47.5% 58.3%, 45.2% 34.5%, 27.5% 76.7%, 0.1% 64.9%, 17.9% 100%, 27.6% 76.8%, 76.1% 97.7%, 74.1% 44.1%)",
                  }}
                />
              </div>

              <div className="mx-auto max-w-2xl text-center">
                <h1 className="text-4xl font-bold tracking-tight text-white sm:text-6xl bg-clip-text text-transparent bg-gradient-to-b from-white to-zinc-400 pb-2">
                  {title}
                </h1>

                {subtitle && (
                  <p className="mt-6 text-lg leading-8 text-zinc-300">{subtitle}</p>
                )}

                {children}

                {(primaryCta || secondaryCta) && (
                  <div className="mt-10 flex items-center justify-center gap-x-6">
                    {primaryCta && (
                      <a
                        href={primaryCta.href}
                        className="rounded-full bg-white px-6 py-3 text-sm font-semibold text-zinc-900 shadow-sm hover:bg-zinc-200 focus-visible:outline focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-white transition-all hover:scale-105"
                      >
                        {primaryCta.label}
                      </a>
                    )}
                    {secondaryCta && (
                      <a
                        href={secondaryCta.href}
                        className="text-sm font-semibold leading-6 text-white hover:text-zinc-200 transition-colors flex items-center gap-2 group"
                      >
                        {secondaryCta.label}
                        <span className="group-hover:translate-x-1 transition-transform">
                          →
                        </span>
                      </a>
                    )}
                  </div>
                )}
              </div>

              {/* Bottom Gradient for smoother integration if needed */}
              <div className="absolute inset-x-0 top-[calc(100%-13rem)] -z-10 transform-gpu overflow-hidden blur-3xl sm:top-[calc(100%-30rem)]">
                <div
                  className="relative left-[calc(50%+3rem)] aspect-[1155/678] w-[36.125rem] -translate-x-1/2 bg-gradient-to-tr from-zinc-200 to-zinc-600 opacity-20 sm:left-[calc(50%+36rem)] sm:w-[72.1875rem]"
                  style={{
                    clipPath:
                      "polygon(74.1% 44.1%, 100% 61.6%, 97.5% 26.9%, 85.5% 0.1%, 80.7% 2%, 72.5% 32.5%, 60.2% 62.4%, 52.4% 68.1%, 47.5% 58.3%, 45.2% 34.5%, 27.5% 76.7%, 0.1% 64.9%, 17.9% 100%, 27.6% 76.8%, 76.1% 97.7%, 74.1% 44.1%)",
                  }}
                />
              </div>
            </div>
          );
        }
        ```
      </Step>

## Props

| Prop           | Type     | Required | Description                               |
| -------------- | -------- | -------- | ----------------------------------------- |
| `title`        | `string` | ✅       | Main heading text                         |
| `subtitle`     | `string` | ❌       | Supporting text below the heading         |
| `primaryCta`   | `object` | ❌       | Primary action button `{ label, href }`   |
| `secondaryCta` | `object` | ❌       | Secondary action button `{ label, href }` |
